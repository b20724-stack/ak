        function startCrafting(itemKey) {
            currentCosmetic = itemKey;
            const data = db[itemKey];
            
            document.getElementById('modal-title').innerText = data.title;
            animBox.innerHTML = data.animHtml;
            chemText.innerHTML = '';
            applyBtn.style.display = 'none';
            modal.style.display = 'flex';

            // 타이핑 효과
            let i = 0;
            const speed = 30; // 타이핑 속도
            const textToType = data.text;
            
            function typeWriter() {
                if (i < textToType.length) {
                    if(textToType.charAt(i) === '<') {
                        let tag = "";
                        while(textToType.charAt(i) !== '>') {
                            tag += textToType.charAt(i);
                            i++;
                        }
                        tag += '>';
                        chemText.innerHTML += tag;
                        i++;
                    } else {
                        chemText.innerHTML += textToType.charAt(i);
                        i++;
                    }
                    setTimeout(typeWriter, speed);
                } else {
                    applyBtn.style.display = 'inline-block';
                }
            }
            setTimeout(typeWriter, 500);
        }

        }

            showDialogue("기본 화장품 <b>[" + name + "]</b>(을)를 캐릭터에 바로 사용했습니다! (생성 과정 생략)");
            triggerSparkle();
        }

        function showDialogue(text) {
            dialogue.innerHTML = text;
            dialogue.style.display = 'block';
            setTimeout(() => { dialogue.style.display = 'none'; }, 4000);
        }

        function triggerSparkle() {
            sparkle.style.opacity = '1';
            setTimeout(() => { sparkle.style.opacity = '0'; }, 1000);
        }
